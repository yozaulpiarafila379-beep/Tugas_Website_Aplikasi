1. Abstract Class EntitasGudang

// ABSTRACT CLASS: EntitasGudang
abstract class EntitasGudang {
  String kode, nama
  abstract void tampilkanInfo()
  void cetakIdentitas()
}

2. Class Barang

// CLASS: Barang extends EntitasGudang
class Barang extends EntitasGudang {
  String kategori, merk, satuan, kondisi
  int jumlahStok
  @override void tampilkanInfo()
  void tambahStok(int jml)
  void kurangiStok(int jml)
}

3. Abstract Class Transaksi

// ABSTRACT CLASS: Transaksi
abstract class Transaksi {
  String idTransaksi, tanggal, kodeBarang
  int jumlah
  abstract void prosesTransaksi()
  void cetakHeader()
}

4. Class BarangMasuk

// CLASS: BarangMasuk extends Transaksi
class BarangMasuk extends Transaksi {
  String namaBarang
  @override void prosesTransaksi() {
    barang = cariBarang(kodeBarang)
    if (barang != null) barang.tambahStok(jumlah)
    else print("❌ Barang tidak ditemukan!")
  }
}

5. Class BarangKeluar

// CLASS: BarangKeluar extends Transaksi
class BarangKeluar extends Transaksi {
  String tujuan

    if (barang != null) barang.kurangiStok(jumlah)
    else print("❌ Barang tidak ditemukan!")
  }
}

6. Class StokBarang

// CLASS: StokBarang
class StokBarang {
  int stokAwal, barangMasuk, barangKeluar
  void hitungStok() {
    stokAkhir = stokAwal + barangMasuk - barangKeluar
    if (stokAkhir == 0) status = "Habis"
    else if (stokAkhir < 5) status = "Menipis"
    else status = "Aman"
  }
}

🧬 4 Pilar OOP

1. Enkapsulasi

Stok dan kondisi barang hanya diubah lewat metode tambahStok() dan kurangiStok().

2. Pewarisan (Inheritance)

Barang extends EntitasGudang. BarangMasuk & BarangKeluar extends Transaksi.

3. Polimorfisme

tampilkanInfo() dan prosesTransaksi() di-override pada turunan.

4. Interaksi Objek

Objek Barang berinteraksi dengan StokBarang dan Transaksi.

                    abstract class EntitasGudang {
                    String kode, nama;
                    EntitasGudang(this.kode, this.nama);
                    void tampilkanInfo();
                    void cetakIdentitas() {
                    print("🔹 $kode — $nama");
                    }
                    }

                    class Barang extends EntitasGudang {
                    String kategori, merk, satuan, kondisi;
                    int jumlahStok;
                    Barang(String kode, String nama, this.kategori, this.merk,
                    this.jumlahStok, this.satuan, this.kondisi) : super(kode, nama);
                    @override void tampilkanInfo() {
                    print("📦 $kode | $nama | $kategori | Stok: $jumlahStok $satuan");
                    }
                    void tambahStok(int jumlah) { jumlahStok += jumlah; }
                    void kurangiStok(int jumlah) {
                    if (jumlahStok >= jumlah) jumlahStok -= jumlah;
                    else print("❌ Stok tidak mencukupi!");
                    }
                    }

6. Tambah Barang

7. Tambah Pelanggan

8. Cari Barang

9. Cari Pelanggan

cariPelanggan(idPelanggan) jika ditemukan → tampilkan data pelanggan jika tidak ditemukan → tampilkan pesan error

10. Transaksi Barang Masuk

input kode barang input jumlah barang input keterangan stok = stok + jumlah totalHarga = hargaBeli × jumlah simpan transaksi masuk

11. Transaksi Barang Keluar

input kode barang input jumlah barang input keterangan jika stok mencukupi stok = stok - jumlah totalHarga = hargaJual × jumlah simpan transaksi keluar jika stok tidak cukup tampilkan pesan error

12. Transaksi Pembelian Pelanggan

input ID pelanggan input kode barang input jumlah barang cek pelanggan cek barang cek stok stok = stok - jumlah totalHarga = hargaJual × jumlah simpan transaksi pembelian

13. Menampilkan Data

tampilkanDaftarBarang() tampilkanDaftarPelanggan() tampilkanDaftarTransaksi()

14. Laporan Stok Akhir

untuk setiap barang nilaiStok = stok × hargaBeli totalNilaiStok += nilaiStok totalUnit += stok tampilkan laporan stok

15. Laporan Keuangan

hitung total pemasukan hitung total pengeluaran labaKotor = pemasukan - pengeluaran persentaseLaba = (labaKotor / pengeluaran) × 100 tampilkan laporan keuangan

2. Pewarisan (Inheritance)

            import 'dart:convert';

            class Barang {
            String kode;
            String nama;
            int stok;
            double hargaBeli;
            double hargaJual;
            String kategori;
            String satuan;
            
            Barang({
            required this.kode,
            required this.nama,
            required this.stok,
            required this.hargaBeli,
            required this.hargaJual,
            this.kategori = 'Umum',
            this.satuan = 'Unit',
            });
            
            Map<String, dynamic> toMap() {
            return {
            'kode': kode,
            'nama': nama,
            'stok': stok,
            'hargaBeli': hargaBeli,
            'hargaJual': hargaJual,
            'kategori': kategori,
            'satuan': satuan,
            };
            }
            
            @override
            String toString() {
            return 'Kode: $kode | Nama: $nama | Stok: $stok $satuan | Harga Beli: Rp${hargaBeli.toStringAsFixed(0)} | Harga Jual: Rp${hargaJual.toStringAsFixed(0)} | Kategori: $kategori';
            }
            }

            class Pelanggan {
            String id;
            String nama;
            String alamat;
            String noTelepon;
            String email;
            String jenisKelamin;
            DateTime tanggalDaftar;
            
            Pelanggan({
            required this.id,
            required this.nama,
            required this.alamat,
            required this.noTelepon,
            this.email = '',
            this.jenisKelamin = '-',
            required this.tanggalDaftar,
            });
            
            Map<String, dynamic> toMap() {
            return {
            'id': id,
            'nama': nama,
            'alamat': alamat,
            'noTelepon': noTelepon,
            'email': email,
            'jenisKelamin': jenisKelamin,
            'tanggalDaftar': tanggalDaftar,
            };
            }
            
            @override
            String toString() {
            return 'ID: $id | Nama: $nama | Alamat: $alamat | No. Telepon: $noTelepon | Email: $email';
            }
            }

            class Transaksi {
            String id;
            String kodeBarang;
            String namaBarang;
            int jumlah;
            String jenis; // 'masuk', 'keluar', 'pembelian'
            DateTime tanggal;
            double totalHarga;
            String idPelanggan;
            String namaPelanggan;
            String keterangan;
            
            Transaksi({
            required this.id,
            required this.kodeBarang,
            required this.namaBarang,
            required this.jumlah,
            required this.jenis,
            required this.tanggal,
            required this.totalHarga,
            this.idPelanggan = '',
            this.namaPelanggan = '',
            this.keterangan = '',
            });
            
            Map<String, dynamic> toMap() {
            return {
            'id': id,
            'kodeBarang': kodeBarang,
            'namaBarang': namaBarang,
            'jumlah': jumlah,
            'jenis': jenis,
            'tanggal': tanggal,
            'totalHarga': totalHarga,
            'idPelanggan': idPelanggan,
            'namaPelanggan': namaPelanggan,
            'keterangan': keterangan,
            };
            }
            
            @override
            String toString() {
            String jenisTeks = '';
            if (jenis == 'masuk') jenisTeks = 'BARANG MASUK';
            else if (jenis == 'keluar') jenisTeks = 'BARANG KELUAR';
            else if (jenis == 'pembelian') jenisTeks = 'PEMBELIAN';
            
            String pelangganInfo = '';
            if (namaPelanggan.isNotEmpty) {
            pelangganInfo = ' | Pelanggan: $namaPelanggan';
            }
            
            return '$jenisTeks: $namaBarang ($kodeBarang) | Jumlah: $jumlah | Total: Rp${totalHarga.toStringAsFixed(0)} | Tanggal: ${tanggal.toString().split(' ')[0]}$pelangganInfo';
            }
            }

            class GudangSystem {
            List<Barang> daftarBarang = [];
            List<Pelanggan> daftarPelanggan = [];
            List<Transaksi> daftarTransaksi = [];
            int counterTransaksi = 1;
            String? inputBuffer;
            int inputIndex = 0;
            
            // ========== METHOD UNTUK INPUT DI DARTPAD ==========
            void setInput(String input) {
            inputBuffer = input;
            inputIndex = 0;
            }
            
            String? readLine() {
            if (inputBuffer == null) return null;
            if (inputIndex >= inputBuffer!.length) return null;
            
            int start = inputIndex;
            while (inputIndex < inputBuffer!.length && inputBuffer![inputIndex] != '\n') {
            inputIndex++;
            }
            String result = inputBuffer!.substring(start, inputIndex);
            if (inputIndex < inputBuffer!.length && inputBuffer![inputIndex] == '\n') {
            inputIndex++;
            }
            return result;
            }
            
            // ========== FUNGSI UTAMA ==========
            
            void tambahBarang(Barang barang) {
            daftarBarang.add(barang);
            }
            
            void tambahPelanggan(Pelanggan pelanggan) {
            daftarPelanggan.add(pelanggan);
            }
            
            Barang? cariBarang(String kode) {
            try {
            return daftarBarang.firstWhere((b) => b.kode == kode);
            } catch (e) {
            return null;
            }
            }
            
            Pelanggan? cariPelanggan(String id) {
            try {
            return daftarPelanggan.firstWhere((p) => p.id == id);
            } catch (e) {
            return null;
            }
            }
            
            bool transaksiMasuk(String kodeBarang, int jumlah, String keterangan) {
            Barang? barang = cariBarang(kodeBarang);
            if (barang == null) {
            print('❌ Barang dengan kode $kodeBarang tidak ditemukan!');
            return false;
            }
            
            double totalHarga = barang.hargaBeli * jumlah;
            barang.stok += jumlah;
            
            Transaksi transaksi = Transaksi(
            id: 'TRX${counterTransaksi.toString().padLeft(4, '0')}',
            kodeBarang: barang.kode,
            namaBarang: barang.nama,
            jumlah: jumlah,
            jenis: 'masuk',
            tanggal: DateTime.now(),
            totalHarga: totalHarga,
            keterangan: keterangan,
            );
            daftarTransaksi.add(transaksi);
            counterTransaksi++;
            
            print('✅ Transaksi masuk berhasil: ${barang.nama} +$jumlah ${barang.satuan}');
            print('   Total harga: Rp${totalHarga.toStringAsFixed(0)}');
            return true;
            }
            
            bool transaksiKeluar(String kodeBarang, int jumlah, String keterangan) {
            Barang? barang = cariBarang(kodeBarang);
            if (barang == null) {
            print('❌ Barang dengan kode $kodeBarang tidak ditemukan!');
            return false;
            }
            
            if (barang.stok < jumlah) {
            print('❌ Stok tidak mencukupi! Stok tersedia: ${barang.stok} ${barang.satuan}');
            return false;
            }
            
            double totalHarga = barang.hargaJual * jumlah;
            barang.stok -= jumlah;
            
            Transaksi transaksi = Transaksi(
            id: 'TRX${counterTransaksi.toString().padLeft(4, '0')}',
            kodeBarang: barang.kode,
            namaBarang: barang.nama,
            jumlah: jumlah,
            jenis: 'keluar',
            tanggal: DateTime.now(),
            totalHarga: totalHarga,
            keterangan: keterangan,
            );
            daftarTransaksi.add(transaksi);
            counterTransaksi++;
            
            print('✅ Transaksi keluar berhasil: ${barang.nama} -$jumlah ${barang.satuan}');
            print('   Total harga: Rp${totalHarga.toStringAsFixed(0)}');
            return true;
            }
            
            bool transaksiPembelian(String kodeBarang, String idPelanggan, int jumlah) {
            Barang? barang = cariBarang(kodeBarang);
            if (barang == null) {
            print('❌ Barang dengan kode $kodeBarang tidak ditemukan!');
            return false;
            }
            
            Pelanggan? pelanggan = cariPelanggan(idPelanggan);
            if (pelanggan == null) {
            print('❌ Pelanggan dengan ID $idPelanggan tidak ditemukan!');
            return false;
            }
            
            if (barang.stok < jumlah) {
            print('❌ Stok tidak mencukupi! Stok tersedia: ${barang.stok} ${barang.satuan}');
            return false;
            }
            
            double totalHarga = barang.hargaJual * jumlah;
            barang.stok -= jumlah;
            
            Transaksi transaksi = Transaksi(
            id: 'TRX${counterTransaksi.toString().padLeft(4, '0')}',
            kodeBarang: barang.kode,
            namaBarang: barang.nama,
            jumlah: jumlah,
            jenis: 'pembelian',
            tanggal: DateTime.now(),
            totalHarga: totalHarga,
            idPelanggan: pelanggan.id,
            namaPelanggan: pelanggan.nama,
            keterangan: 'Pembelian oleh ${pelanggan.nama}',
            );
            daftarTransaksi.add(transaksi);
            counterTransaksi++;
            
            print('✅ Pembelian oleh ${pelanggan.nama}: ${barang.nama} $jumlah ${barang.satuan}');
            print('   Total harga: Rp${totalHarga.toStringAsFixed(0)}');
            return true;
            }
            
            // ========== MENAMPILKAN DATA ==========
            
            void tampilkanBarang() {
            if (daftarBarang.isEmpty) {
            print('📭 Belum ada data barang.');
            return;
            }
            
            print('\n📦 DAFTAR BARANG');
            print('=' * 120);
            print('| Kode    | Nama Barang           | Stok   | Satuan | Harga Beli | Harga Jual | Kategori  |');
            print('=' * 120);
            for (var barang in daftarBarang) {
            String kode = barang.kode.padRight(8);
            String nama = barang.nama.length > 20 ? barang.nama.substring(0, 18) + '..' : barang.nama.padRight(22);
            String stok = barang.stok.toString().padRight(6);
            String satuan = barang.satuan.padRight(6);
            String hBeli = 'Rp${barang.hargaBeli.toStringAsFixed(0)}'.padRight(11);
            String hJual = 'Rp${barang.hargaJual.toStringAsFixed(0)}'.padRight(11);
            String kategori = barang.kategori.padRight(10);
            print('| $kode | $nama | $stok | $satuan | $hBeli | $hJual | $kategori |');
            }
            print('=' * 120);
            print('Total: ${daftarBarang.length} barang\n');
            }
            
            void tampilkanPelanggan() {
            if (daftarPelanggan.isEmpty) {
            print('📭 Belum ada data pelanggan.');
            return;
            }
            
            print('\n👤 DAFTAR PELANGGAN');
            print('=' * 100);
            print('| ID      | Nama Pelanggan      | Alamat              | No. Telepon   | Email          |');
            print('=' * 100);
            for (var pelanggan in daftarPelanggan) {
            String id = pelanggan.id.padRight(8);
            String nama = pelanggan.nama.length > 18 ? pelanggan.nama.substring(0, 16) + '..' : pelanggan.nama.padRight(20);
            String alamat = pelanggan.alamat.length > 18 ? pelanggan.alamat.substring(0, 16) + '..' : pelanggan.alamat.padRight(20);
            String noTelp = pelanggan.noTelepon.padRight(13);
            String email = pelanggan.email.length > 14 ? pelanggan.email.substring(0, 12) + '..' : pelanggan.email.padRight(15);

            String id = transaksi.id.padRight(8);
            String jenis = transaksi.jenis.padRight(14);

            String jumlah = transaksi.jumlah.toString().padRight(6);
            String total = 'Rp${transaksi.totalHarga.toStringAsFixed(0)}'.padRight(11);
            String tanggal = transaksi.tanggal.toString().split(' ')[0].padRight(10);
            String keterangan = transaksi.keterangan.length > 18 ? transaksi.keterangan.substring(0, 16) + '..' : transaksi.keterangan.padRight(20);
            print('| $id | $jenis | $nama | $jumlah | $total | $tanggal | $keterangan |');

            // ========== LAPORAN ==========
            
            void laporanKeuangan() {
            if (daftarTransaksi.isEmpty) {
            print('📭 Belum ada transaksi untuk laporan keuangan.');
            return;
            }
            
            double totalPemasukan = 0;
            double totalPengeluaran = 0;
            int totalBarangMasuk = 0;
            int totalBarangKeluar = 0;
            int totalPembelian = 0;
            
            for (var transaksi in daftarTransaksi) {
            if (transaksi.jenis == 'masuk') {
            totalPengeluaran += transaksi.totalHarga;
            totalBarangMasuk += transaksi.jumlah;
            } else if (transaksi.jenis == 'keluar' || transaksi.jenis == 'pembelian') {
            totalPemasukan += transaksi.totalHarga;
            if (transaksi.jenis == 'keluar') {
            totalBarangKeluar += transaksi.jumlah;
            } else {
            totalPembelian += transaksi.jumlah;
            }
            }
            }
            
            double labaKotor = totalPemasukan - totalPengeluaran;
            double persentaseLaba = totalPengeluaran > 0 ? (labaKotor / totalPengeluaran) * 100 : 0;
            
            print('\n💰 LAPORAN KEUANGAN');
            print('=' * 50);
            print('Periode: ${DateTime.now().toString().split(' ')[0]}');
            print('-' * 50);
            print('📊 RINGKASAN TRANSAKSI:');
            print('  Total Barang Masuk    : $totalBarangMasuk unit');
            print('  Total Barang Keluar   : $totalBarangKeluar unit');
            print('  Total Pembelian       : $totalPembelian unit');
            print('  Total Transaksi       : ${daftarTransaksi.length} transaksi');
            print('-' * 50);
            print('💵 PEMASUKAN:');
            print('  Dari Penjualan        : Rp${totalPemasukan.toStringAsFixed(0)}');
            print('-' * 50);
            print('💸 PENGELUARAN:');
            print('  Untuk Pembelian Stok  : Rp${totalPengeluaran.toStringAsFixed(0)}');
            print('-' * 50);
            print('📈 LABA/RUGI:');
            print('  Laba Kotor            : Rp${labaKotor.toStringAsFixed(0)}');
            print('  Persentase Laba       : ${persentaseLaba.toStringAsFixed(2)}%');
            if (labaKotor > 0) {
            print('  Status                : ✅ UNTUNG');
            } else if (labaKotor < 0) {
            print('  Status                : ❌ RUGI');
            } else {
            print('  Status                : ⚖️ IMPAS');
            }
            print('=' * 50);
            print('');
            }
            
            void laporanStok() {
            if (daftarBarang.isEmpty) {
            print('📭 Belum ada data barang.');
            return;
            }
            
            print('\n📦 LAPORAN STOK AKHIR');
            print('=' * 110);
            print('| Kode    | Nama Barang           | Stok Akhir | Satuan | Nilai Stok (Rp) | Kategori  |');
            print('=' * 110);
            
            double totalNilaiStok = 0;
            int totalUnit = 0;
            
            for (var barang in daftarBarang) {
            double nilaiStok = barang.stok * barang.hargaBeli;
            totalNilaiStok += nilaiStok;
            totalUnit += barang.stok;
            
            String kode = barang.kode.padRight(8);
            String nama = barang.nama.length > 20 ? barang.nama.substring(0, 18) + '..' : barang.nama.padRight(22);
            String stok = barang.stok.toString().padRight(10);
            String satuan = barang.satuan.padRight(6);
            String nilai = 'Rp${nilaiStok.toStringAsFixed(0)}'.padRight(15);
            String kategori = barang.kategori.padRight(10);
            print('| $kode | $nama | $stok | $satuan | $nilai | $kategori |');
            }
            print('=' * 110);
            print('Total Unit Stok  : $totalUnit unit');
            print('Total Nilai Stok : Rp${totalNilaiStok.toStringAsFixed(0)}');
            print('=' * 110);
            print('');
            }
